CREATE TABLE COMMUNITYTAXCERTIFICATE (
    id INT NOT NULL AUTO_INCREMENT,
    CTC_ID INT NOT NULL,
    DATEISSUED DATE,
    CTCNO VARCHAR(50),
    LOCAL_TIN VARCHAR(50),
    BASICTAXDUE DECIMAL(15,2) DEFAULT 0.00,
    BUSTAXAMOUNT DECIMAL(15,2) DEFAULT 0.00,
    BUSTAXDUE DECIMAL(15,2) DEFAULT 0.00,
    SALTAXAMOUNT DECIMAL(15,2) DEFAULT 0.00,
    SALTAXDUE DECIMAL(15,2) DEFAULT 0.00,
    RPTAXAMOUNT DECIMAL(15,2) DEFAULT 0.00,
    RPTAXDUE DECIMAL(15,2) DEFAULT 0.00,
    INTEREST DECIMAL(15,2) DEFAULT 0.00,
    TOTALAMOUNTPAID DECIMAL(15,2) DEFAULT 0.00,
    TRANSDATE DATETIME,
    USERID VARCHAR(50),
    CTCTYPE VARCHAR(20),
    CTCYEAR YEAR,
    INUSE_BV TINYINT(1) DEFAULT 1,
    MUNICIPAL_ID INT,
    PRIMARY KEY (id)
);
